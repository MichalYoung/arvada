# Terminals
None, True, False, INT, IDSTR, STR, +, -, *, //, %, ==, !=, <=, >=, <, >, is, NEWLINE

# Grammar
# The first rule in the grammar specifies the start nonterminal
program := stmt/+

stmt := expr NEWLINE

expr := cexpr

literal := None | True | False | INT | IDSTR | STR

cexpr := literal
  | cexpr bin_op cexpr

bin_op := + | - | * | // | % | == | != | <= | >= | < | > | is
